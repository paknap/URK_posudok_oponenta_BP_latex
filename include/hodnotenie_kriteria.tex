\begin{enumerate}[label=\textbf{\arabic*.}]
    \item \underline{\textbf{Analýza problematiky}} \hfill \underline{\textbf{max. 15}}
    
    \begin{enumerate}
        \item \textit{Prehľad súčasného stavu problematiky a teoretický rozbor} \hfill \textit{max. 10}
        \begin{itemize}
            \item \textbf{Do 10b} – Študent preukázal široký prehľad v danej problematike, vie identifikovať a porovnať  podobné riešenia, vie analyzovať dôvody a dopady riešenia, vie zhromaždiť a analyzovať potrebný teoreticky základ.  
            \item \textbf{Do 7b} – Študent preukázal základný prehľad v danej problematike a analýza súčasného stavu je povrchná. 
            \item \textbf{Do 1b} – Študent nepreukázal v dostatočnej miere, že má prehľad v súčasnom stave problematiky. 
        \end{itemize}
        
        \item \textit{Práca s literárnymi zdrojmi} \hfill \textit{max. 5}
        \begin{itemize}
            \item \textbf{Do 5b} – Boli využité rôzne zdroje informácií (knižné, elektronické, vedecké články...). Pri rozbore alebo zmienke určitého faktu boli použité viaceré zdroje, ktoré podporujú dané tvrdenie. 
            \item \textbf{Do 3b} – V práci bolo využité primerané množstvo rôznorodých zdrojov informácií, ale nie všetky podstatné tvrdenia sa opierali o konkrétny zdroj. 
            \item \textbf{Do 1b} – Vzhľadom na problematiku je množstvo a zloženie zdrojov informácií nedostatočné. 
        \end{itemize}
    \end{enumerate}

    \item \underline{\textbf{Návrh riešenia – Fromulácia obmedzení, postupov a cieľov}} \hfill \underline{\textbf{max. 10}}
    \begin{itemize}
        \item \textbf{Do 10b} – Návrh riešenia obsahuje všetky nasledovné zložky: stanovenie cieľov, opis možných prístupov vedúcich k dosiahnutiu cieľov; zamýšľaný postup riešenia; definície obmedzení, s ktorými riešenie uvažuje. 
        \item \textbf{Do 7b} – Niektorá z uvedených zložiek chýba alebo sú niektoré časti návrhu nejasné či príliš stručné. 
        \item \textbf{Do 3b} – Viaceré z uvedených zložiek chýbajú alebo sú nedostatočné, a teda z úvodnej časti práce nie je zrejmé, ako študent bude riešiť úlohy zadania a čo chce dosiahnuť. 
    \end{itemize}

    \item \underline{\textbf{Realizácia úloh}} \hfill \underline{\textbf{max. 20}}
    \begin{enumerate}
        \item \textit{Názornosť pri dokumentovaní priebehu realizácie} \hfill \textit{max. 10}
        \begin{itemize}
            \item \textbf{Do 10b} – Realizácia je dokumentovaná názorne. V práci sú použité náčrty, diagramy, blokové schémy, prípadne fotografie... 
            \item \textbf{Do 7b} – Určité časti realizácie by si zaslúžili lepšiu názornosť. 
            \item \textbf{Do 3b} – Opis realizácie si od čitateľa vyžaduje vysokú mieru predstavivosti. Nie je jasné ako boli realizované viaceré kroky. 
        \end{itemize}

        \item \textit{Zdôvodnenie postupov a realizovaných krokov} \hfill \textit{max. 5}
        \begin{itemize}
            \item \textbf{Do 5b} – Všetky kroky a postupy pri riešení úloh zadania sú vhodne zdôvodnené a majú zjavné opodstatnenie. 
            \item \textbf{Do 3b} – Niektoré kroky a postupy nie sú zdôvodnené a ani z kontextu nie je zrejmé ich opodstatnenie.
            \item \textbf{Do 1b} – Pri viacerých krokoch a postupoch nie je zrejmý dôvod, prečo boli vykonané. 
        \end{itemize}

        \item \textit{Praktické využitie teoretických poznatkov}  \hfill \textit{max. 5}
        \begin{itemize}
            \item \textbf{Do 5b} – Pri praktickej realizácii boli v dostatočnej miere využité teoretické poznatky. 
            \item \textbf{Do 3b} – Kvalita dosiahnutých výsledkov je znížená z dôvodu nevyužitia niektorých potrebných teoretických poznatkov.
            \item \textbf{Do 1b} – Realizácia úloh nie je postavená na exaktných teoretických poznatkoch napriek tomu, že si to charakter úloh vyžaduje. 
        \end{itemize}
    \end{enumerate}

    \item \underline{\textbf{Prezentácia výsledkov}} \hfill \underline{\textbf{max. 20}}
    \begin{enumerate}
        \item \textit{Funkčnosť riešenia a metodika jej overenia} \hfill \textit{max. 5}
        \begin{itemize}
            \item \textbf{Do 5b} – Z práce je zrejmé, že výsledné riešenie je funkčné alebo spĺňa stanovené požiadavky. Spôsob overenia funkčnosti alebo plnenia požiadaviek je náležite opísaný. 
            \item \textbf{Do 3b} – Z kontextu nie je pochybnosť, že výsledné riešenie aspoň čiastočne spĺňa stanovené požiadavky, ale nie je uvedený spôsob ich overovania. 
            \item \textbf{Do 1b} – Nie je poskytnutý žiaden dôkaz o tom, že výsledné riešenie je funkčné alebo spĺňa stanovené požiadavky. 
        \end{itemize}

        \item \textit{Experimentálne overenie vlastností riešenia} \hfill \textit{max. 10}
        \begin{itemize}
            \item \textbf{Do 10b} – V práci sú zaznamenané a vhodne vizualizované údaje z dostatočného množstva experimentov (tabuľky, inžiniersky exaktné grafy...) s cieľom určiť vlastnosti a obmedzenia realizovaného riešenia. Jednotlivé experimenty sú náležite opísané. 
            \item \textbf{Do 7b} – Bolo vykonané dostatočné množstvo experimentov, ale tie nie sú vhodne prezentované. 
            \item \textbf{Do 3b} – Bolo vykonané nedostatočné množstvo experimentov pre určenie vlastností riešenia alebo kvalita experimentov neumožňuje vyvodiť relevantné závery. 
        \end{itemize}

        \item \textit{Vlastné kvalitatívne zhodnotenie výsledkov} \hfill \textit{max. 5}
        \begin{itemize}
            \item \textbf{Do 5b} – Dosiahnuté výsledky a vlastnosti riešenia sú komplexne zhodnotené a okomentované v širších súvislostiach. 
            \item \textbf{Do 3b} – Študent vyjadril svoj postoj k výsledkom a vykonal základnú analýzu vlastností riešenia. 
            \item \textbf{Do 1b} – Zhodnotenie výsledkov je povrchné alebo nedostatočné. 
        \end{itemize}
    \end{enumerate}

    \item \underline{\textbf{Technická a programová dokumentácia (v prílohe, na proloženom médiu)}} \hfill \underline{\textbf{max. 10}}
    \begin{itemize}
        \item \textbf{Do 10b} – Technická alebo programová dokumentácia dopĺňa prácu o všetky informácie potrebné na nezávislé zopakovanie postupov či experimentov a umožňuje používať a nadviazať na realizované riešenie. Je písaná prehľadne, štruktúrovane a názorne. 
        \item \textbf{Do 7b} – Určité časti technickej dokumentácie nie sú spracované v dostatočnej kvalite na to, aby bolo možné postupy či experimenty presne zopakovať alebo dokumentácia nie je dostatočne prehľadná. 
        \item \textbf{Do 3b} – Technická dokumentácia neposkytuje vhodné doplňujúce informácie. Na prácu nie je možné nadviazať a ani nie je možné zopakovať realizované postupy. 
    \end{itemize}

    \item \underline{\textbf{Celkové hodnotenie}} \hfill \underline{\textbf{max. 15}}
    \begin{enumerate}
        \item \textit{Formálna úroveň} \hfill \textit{max. 5}
        \begin{itemize}
            \item \textbf{Do 5b} – Bola vhodne zvolená štruktúra, rozsah aj formátovanie práce (napr. dodržaním odporúčanej šablóny). Rozloženie a členenie textu, veľkosť obrázkov, tabuliek a medzier medzi jednotlivými prvkami dokumentu pôsobí adekvátne. 
            \item \textbf{Do 3b} – Určité prvky dokumentu sú z formálneho hľadiska použité neadekvátnym, rušivým spôsobom.
            \item \textbf{Do 1b} – Práca po formálnej stránke nepôsobí profesionálne alebo seriózne.
        \end{itemize}

        \item \textit{Úroveň písomného prejavu} \hfill \textit{max. 10}
        \begin{itemize}
            \item \textbf{Do 10b} – Odborné vyjadrovanie je presné a myšlienky sú podávané zrozumiteľne. Text nie je znehodnotený žiadnymi rušivými elementami (preklepy, gramatické či štylistické chyby). 
            \item \textbf{Do 7b} – Odborné vyjadrovanie je v poriadku a myšlienky sú podávané zrozumiteľne, ale text je poznačený viacerými rušivými elementami (preklepy, gramatické či štylistické chyby). 
            \item \textbf{Do 5b} – Odborné vyjadrovanie alebo zrozumiteľné podanie myšlienok má určité nedostatky, ale samotný text je bez rušivých elementov. 
            \item \textbf{Do 3b} – V práci dominuje neodborné vyjadrovanie a niektoré vety či tvrdenia nedávajú zmysel. Častokrát nie je zrejmé, čo chcel autor daným tvrdením vyjadriť. 
        \end{itemize}

        \item \textit{Náročnosť, pôvodnosť a využiteľnosť}     \hfill \textit{max. 10}
        \begin{itemize}
            \item \textbf{Do 10b} – Spôsob riešenia úloh spĺňa požiadavku očakávanej náročnosti vyplývajúcej zo zadania. Práca vykazuje prekryv do 9\ \% s inými autormi z CRZP. Výsledky sú aspoň čiastočne využiteľné aj v budúcich projektoch. 
            \item \textbf{Do 7b} – Realizovaná práca nevykazuje takú náročnosť, aká sa predpokladá zo zadania alebo sú výsledky práce úplne nepoužiteľné v iných projektoch. Práca je však originálna. 
            \item D\textbf{o 4b} – Práca vykazuje neoprávnený prekryv nad 25\ \% s inými autormi z CRZP alebo náročnosť realizácie práce je veľmi nízka. 
        \end{itemize}
    \end{enumerate}
\end{enumerate}
